# frame-master-plugin-react-streaming

Frame-Master plugin

## Installation

```bash
bun add frame-master-plugin-react-streaming
```

## Usage

```typescript
import type { FrameMasterConfig } from "frame-master/server/types";
import framemasterpluginreactstreaming from "frame-master-plugin-react-streaming";

const config: FrameMasterConfig = {
  HTTPServer: { port: 3000 },
  plugins: [framemasterpluginreactstreaming()],
};

export default config;
```
